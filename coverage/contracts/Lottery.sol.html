interface IERC20WithDecimal is IERC20 {

contract Lottery is VRFv2SubscriptionConsumer {
    IERC20WithDecimal private usdtToken;
    address private immutable lotteryOwner;
    uint private immutable maxPlayers;
    uint private currentPlayersCount = 0;

        uint prize;
    }
 
    mapping(uint => address) public players;

    event ResetGame();
    event Winners(LuckyPlayer[]);
 
    constructor(
        address _usdtTokenAddress,
        uint _maxPlayers,
        uint[] memory _prizes,
        uint64 subscriptionId,
        address coordinator,
        bytes32 keyHash
    ) VRFv2SubscriptionConsumer(subscriptionId, coordinator, keyHash) {
        lotteryOwner = msg.sender;
        maxPlayers = _maxPlayers;
        usdtToken = IERC20WithDecimal(_usdtTokenAddress);
        prizes = _prizes;
 
        uint maxTrAmount = MAX_TRANSFER_AMOUNT;
        Eif (_maxPlayers < MAX_TRANSFER_AMOUNT) {
            maxTrAmount = _maxPlayers;
        }
        maxTransferAmount = maxTrAmount;
    }
 
    modifier onlyLotteryOwner() {
        require(msg.sender == lotteryOwner, "Only the contract owner can call this function.");
        _;
    }
 
    function buyLotteryTickets(address player, uint256 amount) public onlyLotteryOwner {
        require(lock == false, "Contract is locked, waiting for results...");
 
        uint256 amountInDollar = amount / (10 ** usdtToken.decimals());
 
        require(1 <= amountInDollar && amountInDollar <= maxTransferAmount, "The amount exceeded the max amount");
        require(currentPlayersCount + amountInDollar <= maxPlayers, "The amount is too much for this cycle");
        require(usdtToken.transferFrom(player, address(this), amount), "USDT transfer failed.");
 
        for (uint i = 1; i <= amountInDollar; i++) {
            players[++currentPlayersCount] = player;
        }
 
        emit NewPlayer(player, amount);
 
        if (currentPlayersCount == maxPlayers) {
            lock = true;
            requestId = requestRandomWords(uint32(prizes.length));
        }
    }
 
    function resetGame() private {
        currentPlayersCount = 0;
        requestId = 0;
        emit ResetGame();
    }
 
    function fulfillRandomWords(uint256 _requestId, uint256[] memory _randomWords) internal override {
        require(requestId == _requestId, "request not found!");
        resetGame();
        transferPrizesToWinners(_randomWords);
    }
 
    function transferPrizesToWinners(uint256[] memory _randomWords) private {
 
        LuckyPlayer[] memory luckyPlayers = new LuckyPlayer[](prizes.length);
 
        for (uint32 i = 0; i < prizes.length; i++) {
            uint luckyPlayerIndex = _randomWords[i] % maxPlayers;
            address luckyPlayer = players[luckyPlayerIndex];
            require(usdtToken.transfer(luckyPlayer, prizes[i]), "USDT transfer failed.");
 
            luckyPlayers[i] = LuckyPlayer({
                playerAddress: luckyPlayer,
                prize: prizes[i]
            });
        }
 
        emit Winners(luckyPlayers);
 
        uint256 contractBalance = usdtToken.balanceOf(address(this));
        require(usdtToken.transfer(lotteryOwner, contractBalance), "USDT transfer to owner failed.");
        lock = false;
    }
}
